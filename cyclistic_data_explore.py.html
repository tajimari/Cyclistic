import numpy as np
import pandas as pd
import cyclistic_data_cleaned as cd

''' 
I did all my cleaning in cyclistic_data_cleaned, then imported the file for data exploration. 
'''

df = cd.df
pd.set_option('display.max_columns', None)

# Find ride count for each group
df.groupby('member_casual')['ride_id'].count()

# Find average ride length for each group to nearest second
round(df.groupby('member_casual')['ride_duration'].mean())

# Difference in ride lengths between casual riders and members by day of week
round(df.groupby(['member_casual', 'day_of_week'])['ride_duration'].mean())

# Calculate how often people use service based on day of week
round(df.groupby(['member_casual', 'day_of_week'])['ride_id'].count())

# Calculate ridership based on month
round(df.groupby(['member_casual', 'month'])['ride_id'].count())

# Tracking ride count throughout the year
df.groupby('date')['ride_id'].count()

# Selecting top 50 start and end stations from members
df[df['member_casual'] == 'member'].groupby('start_station_name')['ride_id'].count().sort_values(ascending=False).head(50)
df[df['member_casual'] == 'member'].groupby('end_station_name')['ride_id'].count().sort_values(ascending=False).head(50)

# Selecting top 50 start and end stations from casual users
df[df['member_casual'] == 'casual'].groupby('start_station_name')['ride_id'].count().sort_values(ascending=False).head(50)
df[df['member_casual'] == 'casual'].groupby('end_station_name')['ride_id'].count().sort_values(ascending=False).head(50)
